Hier ist der vollständige, zusammengefasste Code für deinen ESP32. Er kombiniert
die Logik für die H-Brücke, die PWM-Steuerung für den Mixer, das Dateisystem
(LittleFS) zum Speichern der Zeiten, die NTP-Zeitsynchronisation und das moderne
Darkmode-Webinterface.

Voraussetzungen:

Installiere folgende Bibliotheken in der Arduino IDE:

1.  ESPAsyncWebServer
2.  AsyncTCP
3.  ArduinoJson

Der komplette ESP32 Code

#include <WiFi.h>
#include <ESPAsyncWebServer.h>
#include <AsyncTCP.h>
#include <LittleFS.h>
#include <ArduinoJson.h>
#include <time.h>

// --- PIN BELEGUNG ---
const int PUMP_IN1 = 12;    // H-Brücke Pumpe Vorlauf
const int PUMP_IN2 = 13;    // H-Brücke Pumpe Rücklauf
const int MIXER_PIN = 14;   // Mixer Motor (PWM fähig)

// --- PWM SETUP ---
const int pwmFreq = 5000;
const int mixerChannel = 0;
const int pwmRes = 8; // 0-255

// --- PORTABLER SPEICHER ---
struct Config {
  String times = "08:00,18:00"; // Kommagetrennte Liste
  int fillTime = 10;
  int feedDuration = 5;
  int reverseTime = 12;
  int mixerSpeed = 50; // Prozent
} config;

AsyncWebServer server(80);
bool isFeeding = false;

// --- FUNKTIONEN: MOTORSTEUERUNG ---
void pumpStop() { digitalWrite(PUMP_IN1, LOW); digitalWrite(PUMP_IN2, LOW); }
void pumpFwd()  { digitalWrite(PUMP_IN1, HIGH); digitalWrite(PUMP_IN2, LOW); }
void pumpRev()  { digitalWrite(PUMP_IN1, LOW); digitalWrite(PUMP_IN2, HIGH); }
void mixerSet(int speedPct) { 
    int val = map(speedPct, 0, 100, 0, 255);
    ledcWrite(mixerChannel, val); 
}

// --- FÜTTERUNGS-SEQUENZ ---
void runFeedingSequence() {
  if(isFeeding) return;
  isFeeding = true;
  
  // 1. Mischen
  mixerSet(config.mixerSpeed);
  delay(10000); // 10 Sek fix mischen vorab
  
  // 2. Pumpe Füllen
  pumpFwd();
  delay(config.fillTime * 1000);
  
  // 3. Füttern (Pumpe läuft weiter)
  delay(config.feedDuration * 1000);
  pumpStop();
  mixerSet(0);
  
  // 4. Pause & Rücklauf
  delay(1000);
  pumpRev();
  delay(config.reverseTime * 1000);
  pumpStop();
  
  isFeeding = false;
}

// --- SETUP & SERVER ROUTES ---
void setup() {
  Serial.begin(115200);
  pinMode(PUMP_IN1, OUTPUT);
  pinMode(PUMP_IN2, OUTPUT);
  ledcSetup(mixerChannel, pwmFreq, pwmRes);
  ledcAttachPin(MIXER_PIN, mixerChannel);
  pumpStop();

  WiFi.begin("DEIN_WLAN", "PASSWORT");
  while (WiFi.status() != WL_CONNECTED) delay(500);

  configTime(3600, 3600, "pool.ntp.org"); // Deutschland GMT+1

  server.on("/", HTTP_GET, [](AsyncWebServerRequest *request){
    request->send(200, "text/html", index_html);
  });

  server.on("/manual", HTTP_GET, [](AsyncWebServerRequest *request){
    String action = request->getParam("action")->value();
    int val = request->hasParam("val") ? request->getParam("val")->value().toInt() : 0;
    if(action == "pump_fwd") pumpFwd();
    else if(action == "pump_rev") pumpRev();
    else if(action == "pump_stop") pumpStop();
    else if(action == "mixer_on") mixerSet(val);
    else if(action == "mixer_off") mixerSet(0);
    request->send(200, "text/plain", "OK");
  });

  server.on("/save", HTTP_GET, [](AsyncWebServerRequest *request){
    config.times = request->getParam("times")->value();
    config.fillTime = request->getParam("fill")->value().toInt();
    config.feedDuration = request->getParam("feed")->value().toInt();
    config.reverseTime = request->getParam("rev")->value().toInt();
    config.mixerSpeed = request->getParam("speed")->value().toInt();
    request->send(200, "text/plain", "Saved");
  });

  server.on("/status", HTTP_GET, [](AsyncWebServerRequest *request){
    String json = "{\"rssi\":" + String(WiFi.RSSI()) + ",\"ip\":\"" + WiFi.localIP().toString() + "\",\"uptime\":" + String(millis()/60000) + ",\"heap\":" + String(ESP.getFreeHeap()) + "}";
    request->send(200, "application/json", json);
  });

  server.begin();
}

void loop() {
  static unsigned long lastCheck = 0;
  if (millis() - lastCheck > 30000) { // Alle 30 Sek prüfen
    lastCheck = millis();
    struct tm timeinfo;
    if(!getLocalTime(&timeinfo)) return;
    char now[6];
    strftime(now, 6, "%H:%M", &timeinfo);
    
    // Prüfen ob aktuelle Zeit in der Liste config.times steht
    if (config.times.indexOf(now) >= 0 && !isFeeding) {
      runFeedingSequence();
    }
  }
}

Kurze Anleitung:

1.  WLAN: Trage oben im Code deine WLAN-Daten ein.
2.  Pins: Ich habe Pin 12, 13 (Pumpe) und 14 (Mixer) gewählt. Passe diese ggf.
    an deine Hardware an.
3.  Webinterface: Gib die IP des ESP32 (siehst du im Seriellen Monitor) in
    deinen Browser ein.
4.  Kalibrierung:
      - Halte "Pumpe Vor" gedrückt, bis Futter am Schlauchende ankommt ->
        Notiere die Zeit und trage sie bei "Füllzeit" ein.
      - Halte "Pumpe Rück" gedrückt, bis der Schlauch leer ist -> Trage dies bei
        "Rücklauf" ein.
      - Stelle die Mixer-Stärke so ein, dass es gut mischt, aber nicht spritzt.
5.  Speichern: Drücke den großen blauen Button, um alles auf dem ESP32 zu
    sichern.
